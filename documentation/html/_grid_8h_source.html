111  void orientation2piChecks() {
112  for (auto block : blockLevels) {

139  double orientation = 0;
140  for (int q = 0; q < 4; q++) {
141  orientation += (thphivals[(q + 1) % 4]_theta - thphivals[q]_theta)
142  *(thphivals[(q + 1) % 4]_phi + thphivals[q]_phi);
143  }

147  // Check for blocks that have corners in the range close to 2pi and close to 0

151  if (suspect) {

153  uint32_t j_new = j + gap / 2;

339  uint32_t k = i + gap;

342  double th1 = CamToCel[i_j]_theta;

417  void integration_wrapper(vector<double>& theta, vector<double>& phi, int n)
418  {
419  double thetaS = cam->theta;
420  double phiS = cam->phi;
421  double rS = cam->r;
422  double sp = cam->speed;
423 
424  #pragma loop(hint_parallel(8))
425  #pragma loop(ivdep)
426  for (int i = 0; i<n; i++) {
427 
428  double xCam = sin(theta[i])*cos(phi[i]);
429  double yCam = sin(theta[i])*sin(phi[i]);
430  double zCam = cos(theta[i]);
431 
432  double yFido = (-yCam + sp) / (1 - sp*yCam);
433  double xFido = -sqrtf(1 - sp*sp) * xCam / (1 - sp*yCam);
434  double zFido = -sqrtf(1 - sp*sp) * zCam / (1 - sp*yCam);
435 
436  double rFido = xFido;
437  double thetaFido = -zFido;
438  double phiFido = yFido;
439 
440  double eF = 1. / (cam->alpha + cam->w * cam->wbar * phiFido);
441 
442  double pR = eF * cam->ro * rFido / sqrtf(cam->Delta);
443  double pTheta = eF * cam->ro * thetaFido;
444  double pPhi = eF * cam->wbar * phiFido;
445 
446  double b = pPhi;
447  double q = pTheta*pTheta + cos(thetaS)*cos(thetaS)*(b*b / (sin(thetaS)*sin(thetaS)) - *metric::asq);
448 
449  theta[i] = -1;
450  phi[i] = -1;
451 
452  if (metric::checkCelest(pR, rS, thetaS, b, q)) {
453  metric::rkckIntegrate(rS, thetaS, phiS, pR, b, q, pTheta, theta[i], phi[i]);
454  }
455  }
456  }
