482  void controlPoints(vector<Point2d>& knots, vector<Point2d> &p1, vector<Point2d> &p2) {
483  size_t n = knots.size() - 1;
484  p1.resize(n);
485  p2.resize(n);
486  if (n == 1) {
487  // 3P1 = 2P0 + P3
488  p1[0] = (2. * knots[0] + knots[1]) / 3.;
489  // P2 = 2P1 P0
490  p2[0] = 2 * p1[0] - knots[0];
491  return;
492  }
493  vector<Point2d> r(n);
494  for (int i = 1; i < n - 1; i++)
495  r[i] = 4. * knots[i] + 2. * knots[i + 1];
496  r[0] = knots[0] + 2. * knots[1];
497  r[n - 1] = (8. * knots[n - 1] + knots[n]) / 2.;
498 
499  // First control point
500  solvesystem(r, p1);
501 
502  for (int i = 0; i < n - 1; ++i)
503  p2[i] = 2. * knots[i + 1] - p1[i + 1];
504 
505  p2[n - 1] = (knots[n] + p1[n - 1]) / 2.;
506  };

508  void solvesystem(vector<Point2d> r, vector<Point2d>& x) {
509  size_t n = r.size();
510  vector<double> tmp(n);
511  double b = 2.;
512  x[0] = r[0] / b;
513  for (int i = 1; i < n; i++) {// Decomposition and forward substitution.
514  tmp[i] = 1. / b;
515  b = (i < n - 1 ? 4.0 : 3.5) - tmp[i];
516  x[i] = (r[i] - x[i - 1]) / b;
517  }
518 
519  for (int i = 1; i < n; i++) {// Decomposition and forward substitution.
520  x[n - i - 1] -= tmp[n - i] * x[n - i];
521  }
522 
523  };
