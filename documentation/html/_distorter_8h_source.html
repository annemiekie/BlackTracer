126  void averageAll() {
127  int ch0 = 0;
128  int ch1 = 0;
129  int ch2 = 0;
130  for (int i = 0; i < celestSrc.rows; ++i) {
131  for (int j = 0; j < celestSrc.cols; ++j) {
132  Vec3b col = celestSrc.at<Vec3b>(i, j);
133  ch0 += col.val[0] * col.val[0];
134  ch1 += col.val[1] * col.val[1];
135  ch2 += col.val[2] * col.val[2];
136  }
137  }
138  int numpix = celestSrc.rows*celestSrc.cols;
139  allcol = Vec3b(sqrt(ch0 / numpix), sqrt(ch1 / numpix), sqrt(ch2 / numpix));
140  }

208  bool starInPolygon(Point2d& starPt, vector<Point2d>& thphivals, int sgn) {

489  double redshift(double theta, double phi) {
490  double yCam = sin(theta) * sin(phi);
491  double part = (1. - cam->speed*yCam);
492  double betaPart = sqrt(1 - cam->speed * cam->speed) / part;
493 
494  double yFido = (-yCam + cam->speed) / part;
495  double eF = 1. / (cam->alpha + cam->w * cam->wbar * yFido);
496  double b = eF * cam->wbar * yFido;
497 
498  return 1. / (betaPart * (1 - b*cam->w) / cam->alpha);
499  }

619  double error = 0.00001;
620

622  double starInPixY = 0.5f;

719  phiRight = phiMid;

721  rd = md;
722  lu = (lu + ld)*HALF;
723  }
724  } else {

736  ru = mu;

738  }
739  }
